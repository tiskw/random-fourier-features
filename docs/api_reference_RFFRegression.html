Minimal Example

>>> import numpy as np
>>> import rfflearn.cpu as rfflearn
>>> X = np.array([[-1, -1], [-2, -1], [1, 1], [2, 1]])
>>> y = np.array([1, 1, 2, 2])
>>> reg = rfflearn.RFFRegression().fit(X, y)
>>> reg.score(X, y)
1.0
>>> reg.predict(np.array([[-0.8, -1]]))
array([1.00065865])
